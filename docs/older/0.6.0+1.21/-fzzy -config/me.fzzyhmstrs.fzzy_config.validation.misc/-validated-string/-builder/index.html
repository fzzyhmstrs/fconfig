        if(s.contains("chicken")) {
            ValidationResult.success(s)
        } else {
            if(s.contains("chicken", true)) {
                val s2 = s.replace(Regex("(?i)chicken"), "chicken")

                ValidationResult.error(s, "String must contain the lowercase word 'chicken'")
            }
        }
    }
    .build()
